\section{104. 二叉树的最大深度}
\label{leetcode:104}

\begin{verbatim}
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function(root) {
  return root === null
    ? 0
    : 1 + Math.max(maxDepth(root.left), maxDepth(root.right));
};
\end{verbatim}
