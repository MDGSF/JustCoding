\section{283. 移动零}
\label{leetcode:283}

\subsection{题目}

给定一个数组 nums，编写一个函数将所有 0 移动到数组的末尾，同时保持非零元素的相对顺序。

\textbf{示例}:

\begin{verbatim}
  输入: [0,1,0,3,12]
  输出: [1,3,12,0,0]
\end{verbatim}

\textbf{说明}:

\begin{itemize}
  \item 必须在原数组上操作，不能拷贝额外的数组。
  \item 尽量减少操作次数。
\end{itemize}

\subsection{参考题解}

\begin{verbatim}
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, 
 *   modify nums in-place instead.
 */
var moveZeroes = function(nums) {
  let j = 0;
  for (let i = 0; i < nums.length; i += 1) {
    if (nums[i] !== 0) {
      [nums[i], nums[j]] = [nums[j], nums[i]];
      j += 1;
    }
  }
};
\end{verbatim}
