\section{77. 组合}
\label{leetcode:77}

\subsection{题目}

给定两个整数 n 和 k，返回 \verb|1 ... n| 中所有可能的 k 个数的组合。

\textbf{示例}:

\begin{verbatim}
  输入: n = 4, k = 2
  输出:
  [
    [2,4],
    [3,4],
    [2,3],
    [1,2],
    [1,3],
    [1,4],
  ]
\end{verbatim}

\subsection{参考题解}

\begin{verbatim}
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
var combine = function(n, k) {
  let result = [];
  recursion(n, k, 1, [], result);
  return result;
};

function recursion(n, k, first, cur, result) {
  if (cur.length === k) {
    result.push(cur.slice())
    return;
  }

  for (let i = first; i <= n; i += 1) {
    cur.push(i);
    recursion(n, k, i + 1, cur, result);
    cur.pop();
  }
}
\end{verbatim}
